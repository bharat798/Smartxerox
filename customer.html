import { auth, db, storage } from './firebase-init.js'; import { signInAnonymously, onAuthStateChanged, signInWithCustomToken } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js"; import { doc, getDoc, collection, addDoc, runTransaction, serverTimestamp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js"; import { ref, uploadBytesResumable, getDownloadURL } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-storage.js"; /** * --- Smart Xerox Document Management System --- * Complete Customer Logic with Multi-file support and Tiered Pricing. */ // 1. GLOBAL STATE & CONFIGURATION const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id'; let currentShopId = new URLSearchParams(window.location.search).get('shop'); let user = null; let selectedFiles = []; // Array to store multiple file objects // Shop Rates State (Tiered Pricing) let shopRates = { bwRate: 2, // Standard B&W bwRateBulk: 1.5, // Bulk B&W colorRate: 10, // Standard Color colorRateBulk: 8, // Bulk Color tierThreshold: 3 // Threshold pages (e.g., 3 pages tak standard, upar bulk) }; // Global Settings (Auto-delete, limits) let globalSettings = { maxFileSize: 25, autoDeleteHours: 3 }; // ========================================== // 2. APP INITIALIZATION & AUTH // ========================================== onAuthStateChanged(auth, async (u) => { if (!u) { // Agar user login nahi hai toh anonymous login karein if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) { await signInWithCustomToken(auth, __initial_auth_token); } else { await signInAnonymously(auth); } return; } user = u; console.log("Customer Authenticated:", user.uid); await setupShopContext(); }); /** * Shop ka data aur rates load karne ka function (Dual Path Support) */ async function setupShopContext() { if (!currentShopId) { showAppError("Invalid QR Code", "Kripya dukan ka asli QR code scan karein."); return; } try { // Step A: Load Global Settings const configSnap = await getDoc(doc(db, 'artifacts', appId, 'public', 'data', 'settings', 'global')); if (configSnap.exists()) { globalSettings = { ...globalSettings, ...configSnap.data() }; } // Step B: Load Shop Rates (Try Rule 1 path first, then root) const paths = [ doc(db, 'artifacts', appId, 'public', 'data', 'shops', currentShopId), doc(db, 'shops', currentShopId) ]; let shopDoc = null; for (let path of paths) { const snap = await getDoc(path); if (snap.exists()) { shopDoc = snap; break; } } if (shopDoc) { const data = shopDoc.data(); // Update UI Shop Name const shopNameEl = document.getElementById('ui-shop-name'); const shopAddrEl = document.getElementById('ui-shop-address'); if (shopNameEl) shopNameEl.textContent = data.shopName; if (shopAddrEl) shopAddrEl.textContent = data.address || "Authorized Partner"; // Sync Tiered Rates shopRates = { bwRate: parseFloat(data.bwRate) || 2, bwRateBulk: parseFloat(data.bwRateBulk) || 1.5, colorRate: parseFloat(data.colorRate) || 10, colorRateBulk: parseFloat(data.colorRateBulk) || 8, tierThreshold: parseInt(data.tierThreshold) || 3 }; // Switch from Loading to Main View document.getElementById('loading-state').classList.add('hidden'); document.getElementById('app-main').classList.remove('hidden'); // Check for any active orders in memory checkRecoveryData(); } else { showAppError("Shop Not Found", "Ye dukan hamare system mein registered nahi hai."); } } catch (error) { console.error("Initialization Error:", error); showAppError("Connection Error", "Server se connectivity mein problem hai."); } } // ========================================== // 3. MULTI-FILE SELECTION & DROPZONE // ========================================== const fileInput = document.getElementById('file-input'); const dropzone = document.getElementById('dropzone'); if (fileInput) { fileInput.addEventListener('change', e => { handleFiles(Array.from(e.target.files)); fileInput.value = ''; // Reset input to allow same file re-selection }); } // Drag & Drop Listeners if (dropzone) { ['dragenter', 'dragover'].forEach(e => { dropzone.addEventListener(e, (ev) => { ev.preventDefault(); dropzone.classList.add('drag-active'); }); }); ['dragleave', 'drop'].forEach(e => { dropzone.addEventListener(e, (ev) => { ev.preventDefault(); dropzone.classList.remove('drag-active'); }); }); dropzone.addEventListener('drop', (ev) => { handleFiles(Array.from(ev.dataTransfer.files)); }); } /** * Files ko process karke state mein save karne ka logic */ function handleFiles(files) { files.forEach(file => { // Size Check (25MB limit) if (file.size > globalSettings.maxFileSize * 1024 * 1024) { alert(`File "${file.name}" bahut badi hai (Max ${globalSettings.maxFileSize}MB allowed).`); return; } const isPdf = file.name.toLowerCase().endsWith('.pdf'); // Naya File Object banana const fileObj = { id: Math.random().toString(36).substr(2, 9), fileData: file, fileName: file.name, // Simulated Page Count (Production mein pdf.js use karein) pages: isPdf ? Math.floor(Math.random() * 12) + 1 : 1, settings: { colorMode: 'bw' // Default B&W }, price: 0 }; selectedFiles.push(fileObj); }); renderFileList(); } // ========================================== // 4. PRICING LOGIC (TIERED SYSTEM) // ========================================== /** * 🟢 TIERED PRICING CALCULATION 🟢 * Har file ka price uske pages aur mode ke hisaab se count hoga */ function calculateIndividualPrice(pages, mode) { const threshold = shopRates.tierThreshold; const isStandard = pages <= threshold; if (mode === 'bw') { return isStandard ? (pages * shopRates.bwRate) : (pages * shopRates.bwRateBulk); } else { return isStandard ? (pages * shopRates.colorRate) : (pages * shopRates.colorRateBulk); } } /** * UI Refresh Logic for File List */ function renderFileList() { const listContainer = document.getElementById('file-list'); if (!listContainer) return; listContainer.innerHTML = ""; let overallTotal = 0; selectedFiles.forEach((f) => { // Calculate price for this specific file f.price = calculateIndividualPrice(f.pages, f.settings.colorMode); overallTotal += f.price; // Card Template const card = document.createElement('div'); card.className = "file-card bg-white rounded-3xl p-5 border border-slate-200 shadow-sm flex flex-col md:flex-row md:items-center gap-4 transition-all"; card.innerHTML = `

${f.fileName}

${f.pages} Pages ₹${f.price.toFixed(2)}
`; listContainer.appendChild(card); }); // Handle Empty State & Total const summaryBar = document.getElementById('summary-bar'); const totalEl = document.getElementById('order-total'); if (selectedFiles.length > 0) { summaryBar.classList.remove('hidden'); totalEl.textContent = overallTotal.toFixed(2); } else { summaryBar.classList.add('hidden'); } if (window.lucide) window.lucide.createIcons(); } // Window bindings for dynamic buttons window.updateMode = (id, mode) => { const file = selectedFiles.find(x => x.id === id); if (file) { file.settings.colorMode = mode; renderFileList(); } }; window.removeSelectedFile = (id) => { selectedFiles = selectedFiles.filter(x => x.id !== id); renderFileList(); }; // ========================================== // 5. TOKEN GENERATION (TRANSACTIONAL) // ========================================== async function getNextToken() { const today = new Date().toISOString().split('T')[0]; const counterRef = doc(db, 'artifacts', appId, 'public', 'data', 'counters', `${currentShopId}_${today}`); try { let token = 1; await runTransaction(db, async (t) => { const snap = await t.get(counterRef); if (snap.exists()) { token = snap.data().lastToken + 1; t.update(counterRef, { lastToken: token }); } else { t.set(counterRef, { lastToken: 1, shopId: currentShopId, date: today }); } }); return token; } catch (e) { console.warn("Transaction failed, using fallback token."); return Math.floor(Math.random() * 900) + 100; } } // ========================================== // 6. ORDER SUBMISSION (STORAGE + FIRESTORE) // ========================================== const btnSubmit = document.getElementById('btn-submit'); if (btnSubmit) { btnSubmit.onclick = async () => { if (!user || selectedFiles.length === 0) return; // Button Loading State btnSubmit.disabled = true; btnSubmit.innerHTML = ` Processing Order...`; if (window.lucide) window.lucide.createIcons(); try { const tokenNum = await getNextToken(); const uploadedFilesData = []; // Loop through each file and upload for (let fObj of selectedFiles) { const timestamp = Date.now(); const safeName = fObj.fileName.replace(/[^a-z0-9.]/gi, '_').toLowerCase(); const storagePath = `uploads/${currentShopId}/${timestamp}_${safeName}`; const storageRef = ref(storage, storagePath); // Uploading... const uploadTask = await uploadBytesResumable(storageRef, fObj.fileData); const fileUrl = await getDownloadURL(uploadTask.ref); uploadedFilesData.push({ fileName: fObj.fileName, fileUrl: fileUrl, filePath: storagePath, pages: fObj.pages, price: fObj.price, settings: fObj.settings }); } // Create Master Entry in Prints Collection const orderDoc = { shopId: currentShopId, customerId: user.uid, token: tokenNum, files: uploadedFilesData, billEstimate: document.getElementById('order-total').textContent, notes: document.getElementById('order-notes')?.value || "", status: "Pending", createdAt: serverTimestamp(), expiresAt: Date.now() + (globalSettings.autoDeleteHours * 3600000) }; await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'prints'), orderDoc); // Show Success Modal document.getElementById('final-token').textContent = `#${tokenNum}`; document.getElementById('success-modal').classList.remove('hidden'); // Local storage backup localStorage.setItem('smartXerox_lastOrder', JSON.stringify({ token: tokenNum, timestamp: Date.now(), shop: currentShopId })); } catch (error) { console.error("Submit Error:", error); alert("Order processing failed. Kripya apna internet connection check karein."); btnSubmit.disabled = false; btnSubmit.innerHTML = `Send Order `; if (window.lucide) window.lucide.createIcons(); } }; } // ========================================== // 7. UTILITIES (RECOVERY & ERRORS) // ========================================== function checkRecoveryData() { const saved = localStorage.getItem('smartXerox_lastOrder'); if (saved) { const data = JSON.parse(saved); // Check if data is not older than 3 hours if (Date.now() - data.timestamp < (globalSettings.autoDeleteHours * 3600000)) { const banner = document.getElementById('recovery-banner'); const tokenEl = document.getElementById('rec-token'); if (banner && tokenEl) { tokenEl.textContent = `#${data.token}`; banner.classList.remove('hidden'); } } } } function showAppError(title, msg) { document.getElementById('loading-state').classList.add('hidden'); const errState = document.getElementById('invalid-shop-state'); if (errState) { errState.querySelector('h2').textContent = title; errState.querySelector('p').textContent = msg; errState.classList.remove('hidden'); } } // Initial Icons if (window.lucide) window.lucide.createIcons();
